2 Supplementary Table S2. Exposure definitions

exposure_tbl <- bind_rows(
  tibble(
    component = "Early-Pregnancy Age-Specific Models",
    exposure_timing = "Early pregnancy",
    exposure_code = q4_expos
  ),
  tibble(
    component = "Late-Pregnancy Age-Specific Models",
    exposure_timing = "Late pregnancy",
    exposure_code = q5_expos
  )
) %>%
  mutate(
    exposure_label = pretty_var(exposure_code),
    scale = case_when(
      grepl("DietaryPatterns", exposure_code) ~ "Diet pattern score",
      grepl("DII", exposure_code) ~ "Diet inflammatory index score",
      grepl("IDQ", exposure_code) ~ "Diet quality score",
      TRUE ~ "Continuous percentage or intake metric"
    ),
    linked_use = case_when(
      grepl("1$", exposure_code) ~ "Used in early-pregnancy age-specific, mutual-timing, and early longitudinal attempts",
      grepl("2$", exposure_code) ~ "Used in late-pregnancy age-specific, mutual-timing, and late longitudinal attempts",
      TRUE ~ "Used in prenatal diet modeling"
    )
  ) %>%
  relocate(exposure_label, .after = exposure_code)

make_table(exposure_tbl, caption = "Supplementary Table S2. Exposure definitions and coding")

5 Supplementary Table S5. Candidate associations with q < 0.10

Show code 
candidate_age_specific <- bind_rows(
  early_all %>% mutate(component = paste0("Early pregnancy | ", visit_label(visit))),
  late_all %>% mutate(component = paste0("Late pregnancy | ", visit_label(visit)))
) %>%
  filter(q.value < 0.10) %>%
  transmute(
    component,
    exposure = pretty_var(exposure),
    feature,
    n,
    beta = estimate,
    conf_low = conf.low,
    conf_high = conf.high,
    p_value = p.value,
    q_value = q.value
  )

candidate_mutual <- bind_rows(
  mutual_early_all %>% mutate(component = paste0("Mutual timing early term | ", visit_label(visit))),
  mutual_late_all %>% mutate(component = paste0("Mutual timing late term | ", visit_label(visit)))
) %>%
  filter(q.value < 0.10) %>%
  transmute(
    component,
    exposure = pretty_var(exposure_pair),
    feature,
    n,
    beta = estimate,
    conf_low = conf.low,
    conf_high = conf.high,
    p_value = p.value,
    q_value = q.value
  )

candidate_tbl <- bind_rows(candidate_age_specific, candidate_mutual) %>%
  arrange(q_value, p_value)

make_table(candidate_tbl, caption = "Supplementary Table S5. Candidate associations meeting q < 0.10")

6 Supplementary Table S6. Primary-vs-sensitivity stability summary

Show code 
long_coef_primary <- long_res %>% filter("feature" %in% names(long_res))
long_coef_sens <- long_res_sens %>% filter("feature" %in% names(long_res_sens))

stability_tbl <- bind_rows(
  compare_primary_sensitivity(early_all, early_all_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Early-Pregnancy Age-Specific Models"),
  compare_primary_sensitivity(late_all, late_all_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Late-Pregnancy Age-Specific Models"),
  compare_primary_sensitivity(mutual_early_all, mutual_early_all_sens, c("exposure_pair", "feature", "visit")) %>% mutate(component = "Mutual Timing: Early-Pregnancy Term"),
  compare_primary_sensitivity(mutual_late_all, mutual_late_all_sens, c("exposure_pair", "feature", "visit")) %>% mutate(component = "Mutual Timing: Late-Pregnancy Term"),
  compare_primary_sensitivity(long_coef_primary, long_coef_sens, c("exposure", "feature", "term")) %>% mutate(component = "Longitudinal Interaction Model")
) %>%
  group_by(component) %>%
  summarise(
    matched_rows = n(),
    median_abs_beta_change = median(abs_delta, na.rm = TRUE),
    p90_abs_beta_change = quantile(abs_delta, 0.90, na.rm = TRUE),
    direction_flip_fraction = mean(direction_flip, na.rm = TRUE),
    q_reclassification_fraction = mean(q_reclassified, na.rm = TRUE),
    .groups = "drop"
  )

make_table(stability_tbl, caption = "Supplementary Table S6. Primary-vs-sensitivity stability summary")

status_tbl <- model_status %>%
  mutate(
    component = case_when(
      question == "Q4" ~ "Early-Pregnancy Age-Specific Models",
      question == "Q5" ~ "Late-Pregnancy Age-Specific Models",
      question == "Q6" ~ "Mutual Timing Model",
      question == "Q7" ~ "Longitudinal Interaction Model",
      TRUE ~ question
    ),
    child_age = ifelse(is.na(visit), NA_character_, visit_label(as.integer(visit))),
    exposure = pretty_var(exposure),
    complete_case_n = ifelse(
      is.na(n_complete_min) | !is.finite(n_complete_min),
      NA_character_,
      ifelse(n_complete_min == n_complete_max, as.character(n_complete_min), paste0(n_complete_min, "-", n_complete_max))
    )
  ) %>%
  select(component, analysis, child_age, timing, exposure, term_family, status, reason, complete_case_n, n_features_tested)

make_table(status_tbl, caption = "Supplementary Table S7. Saved execution-status inventory")

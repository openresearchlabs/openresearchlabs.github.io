Figures: maternal diet and child metabolites

Figures from the early-pregnancy, late-pregnancy, mutual-timing, and longitudinal analyses. The longitudinal component now returns coefficient-level output across all attempted exposures, but it still generated substantial execution warnings, so the figures below should be read as supporting context rather than a strong standalone inferential block.

pair_key <- intersect(strip_window_suffix(exp_early), strip_window_suffix(exp_late))

pair_df <- bind_rows(lapply(pair_key, function(k) {
  e <- exp_early[strip_window_suffix(exp_early) == k][1]
  l <- exp_late[strip_window_suffix(exp_late) == k][1]
  if (is.na(e) || is.na(l) || !(e %in% names(cd4)) || !(l %in% names(cd4))) return(NULL)

  data.frame(
    pair = k,
    early = suppressWarnings(as.numeric(cd4[[e]])),

signal_tbl <- bind_rows(
  early_res %>% mutate(component = paste0("Q1 early pregnancy | ", visit_label(visit))) %>% group_by(component) %>% summarise(significant_rows = sum(q.value < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop"),
  late_res %>% mutate(component = paste0("Q2 late pregnancy | ", visit_label(visit))) %>% group_by(component) %>% summarise(significant_rows = sum(q.value < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop"),
  mutual_res %>% mutate(component = paste0("Q3 mutual timing: early term | ", visit_label(visit))) %>% group_by(component) %>% summarise(significant_rows = sum(q_early < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop"),
  mutual_res %>% mutate(component = paste0("Q3 mutual timing: late term | ", visit_label(visit))) %>% group_by(component) %>% summarise(significant_rows = sum(q_late < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop"),
  if (nrow(long_coef_res) > 0) {
    long_coef_res %>%
      group_by(timing) %>%
      summarise(component = paste0("Q4 longitudinal | ", timing), significant_rows = sum(q.value < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop")
  } else {
    data.frame(component = "Q4 longitudinal | no coefficient rows", significant_rows = 0L, total_rows = 0L, stringsAsFactors = FALSE)
  }
) %>%
  mutate(significant_fraction = ifelse(total_rows > 0, significant_rows / total_rows, NA_real_))

signal_tbl %>%
  filter(!is.na(significant_fraction)) %>%
  ggplot(aes(x = fct_reorder(component, significant_fraction), y = significant_fraction, fill = component)) +
  geom_col(show.legend = FALSE) +
  geom_text(
    aes(label = paste0(scales::percent(significant_fraction, accuracy = 0.1), "\n(", significant_rows, "/", total_rows, ")")),
    hjust = -0.05,
    size = 2.9
  ) +
  coord_flip() +
  labs(x = NULL, y = "Fraction of rows with q < 0.10") +
  theme_minimal(base_size = 15)

stability_comp <- bind_rows(
  compare_primary_sensitivity(early_res, early_res_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Q1 early pregnancy age-specific"),
  compare_primary_sensitivity(late_res, late_res_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Q2 late pregnancy age-specific"),
  compare_primary_sensitivity(mutual_early_res, mutual_early_res_sens, c("exposure_pair", "feature", "visit")) %>% mutate(component = "Q3 mutual timing: early term"),
  compare_primary_sensitivity(mutual_late_res, mutual_late_res_sens, c("exposure_pair", "feature", "visit")) %>% mutate(component = "Q3 mutual timing: late term"),
  compare_primary_sensitivity(long_coef_res, long_coef_res_sens, c("exposure", "feature", "term")) %>% mutate(component = "Q4 longitudinal")
)

if (nrow(stability_comp) > 0) {
  stability_tbl <- stability_comp %>%
    group_by(component) %>%
    summarise(
      pairs_compared = n(),
      median_abs_delta = median(abs_delta, na.rm = TRUE),
      q_reclassification_fraction = mean(q_reclassified, na.rm = TRUE),
      .groups = "drop"
    ) %>%
    mutate(
      median_abs_delta = round(median_abs_delta, 3),
      q_reclassification_fraction = round(q_reclassification_fraction, 3)
    )

  stability_tbl %>%
    pivot_longer(cols = c(median_abs_delta, q_reclassification_fraction), names_to = "metric", values_to = "value") %>%
    mutate(metric = recode(metric,
      median_abs_delta = "Median absolute beta change",
      q_reclassification_fraction = "Fraction reclassified at q < 0.10"
    )) %>%
    ggplot(aes(x = component, y = value, fill = component)) +
    geom_col(width = 0.65, show.legend = FALSE) +
    facet_wrap(~ metric, scales = "free_y") +
    labs(x = "Component", y = "Value") +
    theme_minimal(base_size = 15) +
    theme(axis.text.x = element_text(angle = 45, hjust = 1))

7 Age-specific heatmaps

Show code 
all_age_meta <- all_age %>%
  filter(is.finite(q.value)) %>%
  add_feature_meta()

top_exposures <- all_age_meta %>%
  group_by(timing, exposure) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  group_by(timing) %>%
  slice_min(best_q, n = 4, with_ties = FALSE) %>%
  ungroup()

heat_pool <- all_age_meta %>%
  inner_join(top_exposures, by = c("timing", "exposure"))

top_features <- heat_pool %>%
  group_by(timing, feature, feature_group) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  group_by(timing) %>%
  slice_min(best_q, n = 20, with_ties = FALSE) %>%
  ungroup()

heat_df <- heat_pool %>%
  inner_join(top_features, by = c("timing", "feature", "feature_group")) %>%
  mutate(
    panel = paste0(timing, " | ", label_exposure(exposure)),
    feature_lab = paste0(feature_group, " | ", feature)
  )

panel_order <- heat_df %>%
  group_by(panel) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q) %>%
  pull(panel)

for (panel_i in panel_order) {
  panel_df <- heat_df %>%
    filter(panel == panel_i) %>%
    mutate(feature_lab = fct_reorder(feature_lab, q.value, .desc = FALSE))

  p <- panel_df %>%
    ggplot(aes(x = factor(visit), y = feature_lab, fill = estimate)) +
    geom_tile() +
    scale_fill_gradient2(low = "#2166AC", mid = "white", high = "#B2182B", midpoint = 0) +
    labs(
      x = "Child age",
      y = "Metabolite (Group | Feature)",
      fill = "Beta"
    ) +
    theme_minimal(base_size = 18) +
    theme(axis.text.y = element_text(size = 13)) +
    scale_x_discrete(labels = visit_label)

  print(p)
}

# Descriptive trajectories are shown as a companion view to the age-specific results, not as a replacement for longitudinal-model inference.
best_row <- all_age %>%
  filter(is.finite(q.value)) %>%
  arrange(q.value) %>%
  slice_head(n = 1)

if (nrow(best_row) > 0) {
  exp0 <- best_row$exposure[[1]]
  feat0 <- best_row$feature[[1]]

  traj_df <- df_long %>%
    filter(visit_num %in% c(4, 5, 6, 7)) %>%
    mutate(
      exposure_value = suppressWarnings(as.numeric(.data[[exp0]])),
      metabolite_value = suppressWarnings(as.numeric(.data[[feat0]])),
      exposure_tertile = make_exposure_tertiles(exposure_value)
    ) %>%
    filter(is.finite(metabolite_value), !is.na(exposure_tertile)) %>%
    group_by(visit_num, exposure_tertile) %>%
    summarise(
      mean_value = mean(metabolite_value, na.rm = TRUE),
      se = sd(metabolite_value, na.rm = TRUE) / sqrt(dplyr::n()),
      .groups = "drop"
    )

  traj_df %>%
    ggplot(aes(x = factor(visit_num), y = mean_value, color = exposure_tertile, group = exposure_tertile)) +
    geom_line(linewidth = 0.9) +
    geom_point(size = 2) +
    geom_errorbar(aes(ymin = mean_value - 1.96 * se, ymax = mean_value + 1.96 * se), width = 0.1) +
    labs(
      x = "Child age",
      y = "Mean standardized metabolite",
      color = "Exposure tertile"
    ) +
    theme_minimal(base_size = 16) +
    scale_x_discrete(labels = visit_label)

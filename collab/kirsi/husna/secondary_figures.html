Secondary figures: maternal diet and child metabolites

These figures show exposure-level detail, timing-window comparison, and robustness patterns for the age-specific and mutual-timing models. They are not the primary inferential summaries and should not be read as direct evidence for the warning-prone Q7 longitudinal model family unless stated explicitly.

Selected exposures used in this page. These are rule-based appendix exemplars, not hand-picked headline results:

screen_df <- bind_rows(

timing_df <- mutual_res %>%
  add_feature_meta() %>%

4 Figure B4. Age-pattern comparison across child visits

Show code 
base_early <- unique(strip_window_suffix(early_res$exposure))
base_late <- unique(strip_window_suffix(late_res$exposure))
common_base <- intersect(base_early, base_late)

base_rank <- bind_rows(
  early_res %>% mutate(base = strip_window_suffix(exposure)),
  late_res %>% mutate(base = strip_window_suffix(exposure))
) %>%
  filter(base %in% common_base) %>%
  group_by(base) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q)

selected_base <- if (nrow(base_rank) > 0) base_rank$base[[1]] else NA_character_

exp_early <- early_res %>%
  filter(strip_window_suffix(exposure) == selected_base) %>%
  group_by(exposure) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q) %>%
  slice_head(n = 1) %>%
  pull(exposure)

exp_late <- late_res %>%
  filter(strip_window_suffix(exposure) == selected_base) %>%
  group_by(exposure) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q) %>%
  slice_head(n = 1) %>%
  pull(exposure)

traj_df <- bind_rows(
  early_res %>% filter(exposure == exp_early) %>% mutate(window = "Early pregnancy"),
  late_res %>% filter(exposure == exp_late) %>% mutate(window = "Late pregnancy")
) %>%
  add_feature_meta()

top_traj_feat <- traj_df %>%
  group_by(feature) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  slice_min(best_q, n = 8, with_ties = FALSE)

traj_plot <- traj_df %>%
  semi_join(top_traj_feat, by = "feature") %>%
  mutate(
    visit_age = factor(visit, levels = c(4, 5, 6, 7), labels = c("6 mo", "12 mo", "24 mo", "5-6 y")),
    feature_lab = paste0(feature_group, " | ", feature)
  )

traj_plot %>%
  ggplot(aes(x = visit_age, y = estimate, color = window, group = window)) +
  geom_hline(yintercept = 0, linetype = "dashed", color = "grey45") +
  geom_line(linewidth = 0.8) +
  geom_point(size = 1.8) +
  geom_errorbar(aes(ymin = conf.low, ymax = conf.high), width = 0.1, linewidth = 0.35) +
  facet_wrap(~ feature_lab, scales = "free_y") +
  labs(
    x = "Child age panel",
    y = "Standardized beta (95% CI)",
    color = "Pregnancy window",
    title = paste0("Age-specific longitudinal comparison for exposure family: ", selected_base)
  ) +
  theme_minimal(base_size = 15)

5 Figure B5. Sensitivity stability by window and visit

Show code 
stability_tbl <- bind_rows(
  compare_primary_sensitivity(early_res, early_res_sens, c("exposure", "feature", "visit")) %>% mutate(window = "Early pregnancy"),
  compare_primary_sensitivity(late_res, late_res_sens, c("exposure", "feature", "visit")) %>% mutate(window = "Late pregnancy")
) %>%
  group_by(window, visit) %>%
  summarise(
    pairs_compared = n(),
    median_abs_delta = median(abs_delta, na.rm = TRUE),
    q_reclassification_fraction = mean(q_reclassified, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    median_abs_delta = round(median_abs_delta, 3),
    q_reclassification_fraction = round(q_reclassification_fraction, 3),
    visit_lab = factor(visit, levels = c(4, 5, 6, 7), labels = c("6 mo", "12 mo", "24 mo", "5-6 y"))
  )

make_table(stability_tbl, caption = "Age-specific sensitivity stability summary")

6 Figure B6. Metabolite-class composition of significant hits

Show code 
class_comp <- bind_rows(
  early_res %>% filter(q.value < 0.1) %>% mutate(component = paste0("Q4 early | visit ", visit)),
  late_res %>% filter(q.value < 0.1) %>% mutate(component = paste0("Q5 late | visit ", visit)),
  mutual_res %>% filter(q_early < 0.1) %>% mutate(component = paste0("Q6 early term | visit ", visit)),
  mutual_res %>% filter(q_late < 0.1) %>% mutate(component = paste0("Q6 late term | visit ", visit))
) %>%
  add_feature_meta() %>%
  count(component, feature_group, name = "n_hits") %>%
  group_by(component) %>%
  mutate(hit_fraction = n_hits / sum(n_hits)) %>%
  ungroup()

make_table(
  class_comp %>% mutate(hit_fraction = round(hit_fraction, 3)),
  caption = "Metabolite-class composition of significant Husna hits"
)
Show code 
class_comp %>%
  ggplot(aes(x = component, y = hit_fraction, fill = feature_group)) +
  geom_col(position = "fill", width = 0.7) +

  labs(
    x = NULL,
    y = "Share of significant hits",
    fill = "Metabolite class",
    title = "Class composition of significant Husna findings"
  ) +
  theme_minimal(base_size = 15)

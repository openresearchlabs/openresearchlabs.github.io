Secondary figures: maternal diet and child metabolites

These figures show exposure-level detail, timing-window comparison, and robustness patterns for the age-specific and mutual-timing models. They are not the primary inferential summaries and should not be read as direct evidence for the warning-prone Q4 longitudinal model family unless stated explicitly.

1 Part 1. Individual Metabolite-Level Secondary Figures (Primary Analysis)

These figures display results at the individual metabolite level, grouped and colored by the 5 predefined biological classifications (Inflammation, Amino Acids, Glycolysis & Energy, Fatty Acids, and Lipoproteins & Lipids).

1.4 Figure B4. Age-pattern comparison across child visits

Show code 
base_early <- unique(strip_window_suffix(early_res$exposure))
base_late <- unique(strip_window_suffix(late_res$exposure))
common_base <- intersect(base_early, base_late)

base_rank <- bind_rows(
  early_res %>% mutate(base = strip_window_suffix(exposure)),
  late_res %>% mutate(base = strip_window_suffix(exposure))
) %>%
  filter(base %in% common_base) %>%
  group_by(base) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q)

selected_base <- if (nrow(base_rank) > 0) base_rank$base[[1]] else NA_character_

exp_early <- early_res %>%
  filter(strip_window_suffix(exposure) == selected_base) %>%
  group_by(exposure) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q) %>%
  slice_head(n = 1) %>%
  pull(exposure)

exp_late <- late_res %>%
  filter(strip_window_suffix(exposure) == selected_base) %>%
  group_by(exposure) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q) %>%
  slice_head(n = 1) %>%
  pull(exposure)

traj_df <- bind_rows(
  early_res %>% filter(exposure == exp_early) %>% mutate(window = "Early pregnancy"),
  late_res %>% filter(exposure == exp_late) %>% mutate(window = "Late pregnancy")
) %>%
  add_feature_meta()

top_traj_feat <- traj_df %>%
  group_by(feature) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  slice_min(best_q, n = 8, with_ties = FALSE)

traj_plot <- traj_df %>%
  semi_join(top_traj_feat, by = "feature") %>%
  mutate(
    visit_age = factor(visit, levels = c(4, 5, 6, 7), labels = c("6 mo", "12 mo", "24 mo", "5-6 y")),
    feature_lab = paste0(feature_group, " | ", feature)
  )

traj_plot %>%
  ggplot(aes(x = visit_age, y = estimate, color = window, group = window)) +
  geom_hline(yintercept = 0, linetype = "dashed", color = "grey45") +
  geom_line(linewidth = 0.8) +
  geom_point(size = 1.8) +
  geom_errorbar(aes(ymin = conf.low, ymax = conf.high), width = 0.1, linewidth = 0.35) +
  facet_wrap(~ feature_lab, scales = "free_y") +

stability_tbl <- bind_rows(

2.4 Figure B11. Sex-stratified effects for significant Diet × Sex interactions

Sex-stratified analyses are conducted and presented separately in boys (blue) and girls (red) ONLY for outcome-exposure pairs that demonstrated a significant Diet × Sex interaction (\(p_{\text{interaction}} < 0.05\)).

Show code 
strat_df_sec <- bind_rows(
  early_res_sec %>% mutate(window = "Early pregnancy"),
  late_res_sec %>% mutate(window = "Late pregnancy")
) %>%
  filter(!is.na(p_interaction) & p_interaction < 0.05)

if (nrow(strat_df_sec) == 0) {
  ggplot() +
    annotate("text", x = 0, y = 0, label = "No significant Diet x Sex interactions (p < 0.05) found in group-level models.") +
    theme_void()
} else {
  strat_long_sec <- strat_df_sec %>%
    mutate(
      label = paste0(visit, "mo | ", label_group(feature), "\nvs. ", label_exposure_detailed(exposure), " (", window, ")")
    ) %>%
    select(label, estimate_boys, conf.low_boys, conf.high_boys, estimate_girls, conf.low_girls, conf.high_girls) %>%
    pivot_longer(
      cols = -label,
      names_to = c(".value", "sex"),
      names_pattern = "(.*)_(boys|girls)"
    ) %>%
    mutate(
      sex = factor(sex, levels = c("boys", "girls"), labels = c("Boys", "Girls"))
    )
  
  ggplot(strat_long_sec, aes(x = estimate, y = label, color = sex)) +
    geom_vline(xintercept = 0, linetype = "dashed", color = "grey50") +
    geom_errorbarh(aes(xmin = conf.low, xmax = conf.high), height = 0.2, linewidth = 0.9, position = position_dodge(0.4)) +
    geom_point(size = 3.5, position = position_dodge(0.4)) +
    scale_color_manual(values = c("Boys" = "#2c7bb6", "Girls" = "#d7191c")) +
    theme_minimal(base_size = 13) +
    theme(
      legend.position = "right",
      axis.title.y = element_blank()
    ) +
    labs(
      x = "Sex-stratified standardized beta (95% CI)",
      color = "Child sex",
      title = "Maternal diet associations stratified by child sex (biological groups)"
    )
}

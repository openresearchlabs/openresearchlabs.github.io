show_table <- function(tbl, page_len = 10, caption = NULL, title = NULL) {
  if (!has_DT) return(tbl)
  tbl_dt <- DT::datatable(
    tbl,
    caption = caption,
    rownames = FALSE,
    filter = "top",
    options = list(
      pageLength = page_len,
      scrollX = FALSE,
      autoWidth = FALSE
    )
  )

  if (!is.null(title) && nzchar(title)) {
    return(htmltools::tagList(htmltools::tags$h4(title), tbl_dt))
  }

  tbl_dt
}

3 Covariate selection framework

Covariate selection is pre-specified and DAG-informed:

  1. The primary adjustment set is fixed in advance.
  2. Balance tests and missingness summaries are reported.
  3. Secondary models add sensitivity covariates.

4 Group balance tests across intervention arms

Covariate balance diagnostics across intervention groups:

  • Kruskal-Wallis for continuous covariates.
  • Fisher exact for discrete covariates (ordinal and nominal categorical).
  • BH adjusted p-values across all visits in the combined table.

These p-values are not used alone to choose adjustment covariates.

run_covariate_test <- function(df, group_var, covariate, test_method) {
  out <- data.frame(
    var = covariate,
    test = test_method,
    p_value = NA_real_,
    p_adj_bh = NA_real_,
    n_groups = NA_integer_,
    note = NA_character_,
    stringsAsFactors = FALSE
  )

  if (!covariate %in% names(df)) {
    out$note <- "covariate_missing_in_data"
    return(out)
  }
  if (!group_var %in% names(df)) {
    out$note <- "group_var_missing_in_data"
    return(out)
  }
  if (!test_method %in% c("anova", "kruskal_wallis", "fisher_or_chisq")) {
    out$note <- "invalid_test_method"
    return(out)
  }

  x <- df[[covariate]]
  g <- df[[group_var]]
  keep <- !is.na(x) & !is.na(g)
  if (!any(keep)) {
    out$note <- "no_complete_cases"
    return(out)
  }

  x <- x[keep]
  g <- as.factor(g[keep])
  out$n_groups <- nlevels(g)

  if (out$n_groups < 2L) {
    out$note <- "fewer_than_two_groups"
    return(out)
  }
  if (length(unique(x)) < 2L) {
    out$note <- "constant_covariate"
    return(out)
  }

  if (identical(test_method, "anova")) {
    out$p_value <- tryCatch({
      fit <- aov(as.numeric(x) ~ g)
      summary(fit)[[1]][["Pr(>F)"]][1]
    }, error = function(e) NA_real_)
  } else if (identical(test_method, "kruskal_wallis")) {
    out$test <- "kruskal_wallis"
    y <- if (is.character(x) || is.factor(x)) as.numeric(factor(x)) else as.numeric(x)
    out$p_value <- tryCatch(kruskal.test(y ~ g)$p.value, error = function(e) NA_real_)
  } else {
    x_cat <- as.factor(x)
    if (nlevels(x_cat) < 2L) {
      out$note <- "fewer_than_two_levels"
      return(out)
    }

    tbl <- table(g, x_cat)
    if (any(dim(tbl) < 2L)) {
      out$note <- "invalid_contingency_table"
      return(out)
    }

    out$p_value <- tryCatch({
      out$test <- "fisher_exact"
      fisher.test(tbl)$p.value
    }, error = function(e) NA_real_)
  }

  out$note <- ifelse(is.na(out$p_value), "test_failed", "ok")
  out
}

run_group_balance_panel <- function(df, group_var, covariates, panel_name) {
  res <- bind_rows(lapply(seq_len(nrow(covariates)), function(i) {
    run_covariate_test(
      df = df,
      group_var = group_var,
      covariate = covariates$var[i],
      test_method = covariates$test[i]
    )
  }))
  res$panel <- panel_name
  res %>% arrange(p_value)
}

pretty_test <- function(x) dplyr::case_when(
  x == "anova" ~ "ANOVA",
  x == "kruskal_wallis" ~ "Kruskal-Wallis",
  x == "fisher_exact" ~ "Fisher exact",
  TRUE ~ x
)
format_p <- function(x) ifelse(is.na(x), "", formatC(x, format = "fg", digits = 4))
format_p2 <- function(x) ifelse(is.na(x), "", formatC(x, format = "f", digits = 2))

make_balance_display <- function(tbl) {
  visit_label <- function(panel_name) {
    gsub("visit", "visit ", panel_name)

  tbl %>%
    filter(note == "ok", !is.na(p_adj_bh)) %>%

      visit = visit_label(panel),
      var = var,
      test = pretty_test(test),
      p = format_p(p_value),

    )
}

plot_balance_var <- function(df, group_var, var_name, test_name) {
  x <- df[[var_name]]
  g <- as.factor(df[[group_var]])
  keep <- !is.na(x) & !is.na(g)

  if (!any(keep)) return(NULL)

  dat <- data.frame(
    group = as.factor(g[keep]),
    value = x[keep],
    stringsAsFactors = FALSE
  )

  if (test_name %in% c("anova", "kruskal_wallis")) {
    dat$value <- suppressWarnings(as.numeric(dat$value))
    ggplot(dat, aes(x = group, y = value)) +
      geom_boxplot(outlier.alpha = 0.4, width = 0.6) +
      geom_jitter(width = 0.12, alpha = 0.35, size = 1.2) +
      labs(
        title = paste0(var_name, " (", pretty_test(test_name), ")"),
        x = "Intervention",
        y = var_name
      ) +
      theme_minimal(base_size = 11)
  } else {
    dat$value <- as.factor(dat$value)
    ggplot(dat, aes(x = group, fill = value)) +
      geom_bar(position = "fill") +
      labs(
        title = paste0(var_name, " (", pretty_test(test_name), ")"),
        x = "Intervention",
        y = "Proportion",
        fill = var_name
      ) +
      theme_minimal(base_size = 11)
  }
}

top_by_panel <- function(tbl, n = 2L) {
  tbl %>%
    filter(note == "ok", !is.na(p_value)) %>%
    arrange(p_value) %>%
    slice_head(n = n)
}

build_prop_plot_data <- function(df, group_var, var_name, panel_title) {
  x <- df[[var_name]]
  g <- as.factor(df[[group_var]])
  keep <- !is.na(x) & !is.na(g)

  data.frame(

  )
}

5 Symbol key

  • \(E\): exposure
  • \(Y\): outcome
  • \(C_k\): covariate \(k\)
  • \(i\): participant index
  • \(a\): age/timepoint index
  • \(t\): repeated-measure time index

6 Causal roles

  • Confounder: common cause of E and Y; adjust in primary models.
  • Mediator: on path E -> mediator -> Y; do not include in primary confounder set.
  • Collider: common effect of two causes; do not condition on it.

11 Proposed models

  1. Age-specific linear model: Y_age ~ E + maternal_core_covariates + visit_specific_fasting
  2. Mutually adjusted timing model: Y_age ~ E_early + E_late + maternal_core_covariates + visit_specific_fasting
  3. Repeated-measures mixed model: Y_it ~ E + age_t + E:age_t + maternal_core_covariates + (1|child)
  4. Sensitivity models: add sensitivity covariate block and compare estimates.

12 Mathematical model forms

  1. Age-specific linear model (age a):

\[ Y_{i,a} = \beta_0 + \beta_1 E_i + \sum_{k=1}^{p} \gamma_k C_{i,k} + \delta F_{i,a} + \varepsilon_{i,a} \]

  1. Mutually adjusted timing model (age a):

\[ Y_{i,a} = \beta_0 + \beta_1 E_{i,\mathrm{early}} + \beta_2 E_{i,\mathrm{late}} + \sum_{k=1}^{p} \gamma_k C_{i,k} + \delta F_{i,a} + \varepsilon_{i,a} \]

  1. Repeated-measures mixed model (child i, time t):

\[ Y_{i,t} = \beta_0 + \beta_1 E_i + \beta_2 t + \beta_3(E_i \times t) + \sum_{k=1}^{p} \gamma_k C_{i,k} + u_i + \varepsilon_{i,t} \]

with random intercept:

\[ u_i \sim \mathcal{N}(0, \sigma_u^2), \quad \varepsilon_{i,t} \sim \mathcal{N}(0, \sigma^2) \]

build_variable_key <- function(df) {
  nm <- names(df)
  make_friendly <- function(x) {
    y <- gsub("([a-z])([A-Z])", "\\1 \\2", x)
    y <- gsub("([A-Za-z])([0-9])", "\\1 \\2", y)
    y <- gsub("([0-9])([A-Za-z])", "\\1 \\2", y)
    y <- gsub("[._]+", " ", y)
    y <- gsub("\\s+", " ", trimws(y))
    tools::toTitleCase(tolower(y))
  }
  var_label <- sapply(nm, function(v) {
    lb <- attr(df[[v]], "label")
    if (is.null(lb) || is.na(lb) || lb == "") "" else as.character(lb)
  })
  value_levels <- sapply(nm, function(v) {
    labs <- attr(df[[v]], "labels")
    if (is.null(labs) || length(labs) == 0) return("")
    paste(sprintf("%s = %s", names(labs), unname(labs)), collapse = "; ")
  })
  data.frame(
    Friendly_Name = sapply(nm, make_friendly),
    Variable_Code = nm,
    Variable_Label = var_label,
    Value_Levels = value_levels,
    stringsAsFactors = FALSE
  )
}

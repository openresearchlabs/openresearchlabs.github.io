3.2 Group balance tests across intervention arms

Why this matters: Balance Diagnostics

These tests check whether covariates are balanced across the intervention arms. Because this study evaluates a randomized trial context, minor imbalances can arise by chance. Note that standard guidelines discourage using p-values alone to drop covariates—pre-specified confounders are retained regardless of balance test significance.

Covariate balance diagnostics across intervention groups:

  • Kruskal-Wallis for continuous covariates.
  • Fisher exact for discrete covariates (ordinal and nominal categorical).
  • BH adjusted p-values across all visits in the combined table.

These p-values are not used alone to choose adjustment covariates. The intervention variable itself is the grouping variable and is therefore not included as a balance-test covariate.

run_covariate_test <- function(df, group_var, covariate, test_method) {
  out <- data.frame(
    var = covariate,
    test = test_method,
    p_value = NA_real_,
    p_adj_bh = NA_real_,
    n_groups = NA_integer_,
    note = NA_character_,
    stringsAsFactors = FALSE
  )

  if (!covariate %in% names(df)) {
    out$note <- "covariate_missing_in_data"
    return(out)
  }
  if (!group_var %in% names(df)) {
    out$note <- "group_var_missing_in_data"
    return(out)
  }
  if (!test_method %in% c("anova", "kruskal_wallis", "fisher_or_chisq")) {
    out$note <- "invalid_test_method"
    return(out)
  }

  x <- df[[covariate]]
  g <- df[[group_var]]
  keep <- !is.na(x) & !is.na(g)
  if (!any(keep)) {
    out$note <- "no_complete_cases"
    return(out)
  }

  x <- x[keep]
  g <- as.factor(g[keep])
  out$n_groups <- nlevels(g)

  if (out$n_groups < 2L) {
    out$note <- "fewer_than_two_groups"
    return(out)
  }
  if (length(unique(x)) < 2L) {
    out$note <- "constant_covariate"
    return(out)
  }

  if (identical(test_method, "anova")) {
    out$p_value <- tryCatch({
      fit <- aov(as.numeric(x) ~ g)
      summary(fit)[[1]][["Pr(>F)"]][1]
    }, error = function(e) NA_real_)
  } else if (identical(test_method, "kruskal_wallis")) {
    out$test <- "kruskal_wallis"
    y <- if (is.character(x) || is.factor(x)) as.numeric(factor(x)) else as.numeric(x)
    out$p_value <- tryCatch(kruskal.test(y ~ g)$p.value, error = function(e) NA_real_)
  } else {
    x_cat <- as.factor(x)
    if (nlevels(x_cat) < 2L) {
      out$note <- "fewer_than_two_levels"
      return(out)
    }

    tbl <- table(g, x_cat)
    if (any(dim(tbl) < 2L)) {
      out$note <- "invalid_contingency_table"
      return(out)
    }

    out$p_value <- tryCatch({
      out$test <- "fisher_exact"
      fisher.test(tbl)$p.value
    }, error = function(e) NA_real_)
  }

  out$note <- ifelse(is.na(out$p_value), "test_failed", "ok")
  out
}

3.4 Causal roles

  • Confounder: common cause of E and Y; adjust in primary models.
  • Mediator: on path E -> mediator -> Y; do not include in primary confounder set.
  • Collider: common effect of two causes; do not condition on it.

3.7 Collinearity

Understanding Collinearity & VIF
  • Pairwise Screen: A Spearman correlation, Epsilon-squared, or Cramer’s V \(\ge 0.70\) highlights redundant covariate pairs. In such cases, standard errors of the exposure effect estimate may inflate if both are kept.
  • Model-Matrix VIF: Variance Inflation Factors (VIF) summarize design-wide multicollinearity. Values below 5 indicate no concern. Values above 5 (or rank deficiency indicating exact aliasing) require reviewing the models for redundant parameters or category sparseness.

Collinearity is checked before interpreting expanded adjustment sets:

  • Numeric covariate pairs are screened with Spearman correlations within each visit-linked panel.
  • Categorical covariate pairs are screened with Cramer’s V.
  • Mixed continuous-categorical pairs are screened with Epsilon-squared from Kruskal-Wallis tests.
  • Pairs with absolute values at or above 0.70 are flagged for review.

cramers_v <- function(x, y) {
  keep <- !is.na(x) & !is.na(y)
  x <- as.factor(x[keep])
  y <- as.factor(y[keep])
  if (length(x) == 0 || nlevels(x) < 2L || nlevels(y) < 2L) return(NA_real_)
  tab <- table(x, y)
  chi <- suppressWarnings(chisq.test(tab, correct = FALSE))
  n <- sum(tab)
  denom <- n * (min(dim(tab)) - 1)
  if (!is.finite(denom) || denom <= 0) return(NA_real_)
  sqrt(as.numeric(chi$statistic) / denom)
}

kruskal_epsilon_sq <- function(statistic, n, groups) {
  if (!is.finite(statistic) || !is.finite(n) || !is.finite(groups) || n <= groups) return(NA_real_)
  max(0, as.numeric((statistic - groups + 1) / (n - groups)))
}

collinearity_screen <- function(df, vars, panel_name, threshold = 0.70) {
  vars <- intersect(vars, names(df))
  vars <- vars[vapply(vars, function(v) sum(!is.na(df[[v]])) >= 10, logical(1))]
  if (length(vars) < 2L) return(data.frame())

  is_continuous <- function(val) {
    if (is.character(val) || is.factor(val) || is.logical(val) || inherits(val, "haven_labelled")) return(FALSE)
    val_num <- suppressWarnings(as.numeric(val))
    sum(is.finite(val_num)) >= 8 && length(unique(val_num[is.finite(val_num)])) >= 6
  }

  pairs <- utils::combn(vars, 2, simplify = FALSE)
  bind_rows(lapply(pairs, function(pair) {
    x <- df[[pair[[1]]]]
    y <- df[[pair[[2]]]]
    
    keep <- !is.na(x) & !is.na(y)
    n_complete <- sum(keep)
    if (n_complete < 10) return(NULL)
    
    x_cont <- is_continuous(x)
    y_cont <- is_continuous(y)
    
    if (x_cont && y_cont) {
      value <- suppressWarnings(cor(as.numeric(x), as.numeric(y), use = "pairwise.complete.obs", method = "spearman"))
      metric <- "spearman_rho"
      flag <- is.finite(value) && abs(value) >= threshold
    } else if (!x_cont && !y_cont) {
      value <- cramers_v(x, y)
      metric <- "cramers_v"
      flag <- is.finite(value) && value >= threshold
    } else {
      # Mixed continuous-categorical pair
      if (x_cont) {
        continuous_val <- as.numeric(x)
        categorical_val <- factor(as.character(y))
      } else {
        continuous_val <- as.numeric(y)
        categorical_val <- factor(as.character(x))
      }
      
      if (nlevels(categorical_val) < 2) return(NULL)
      tst <- suppressWarnings(kruskal.test(continuous_val ~ categorical_val))
      value <- kruskal_epsilon_sq(tst$statistic, n_complete, nlevels(categorical_val))
      metric <- "epsilon_sq"
      flag <- is.finite(value) && value >= threshold
    }

    data.frame(
      panel = panel_name,
      covariate_1 = pair[[1]],
      covariate_2 = pair[[2]],
      metric = metric,
      value = value,
      flagged = flag,
      stringsAsFactors = FALSE
    )
  })) %>%
    filter(is.finite(value)) %>%
    arrange(desc(abs(value)))
}

3.8 Model-matrix collinearity

Model-matrix collinearity is checked before interpreting expanded adjustment sets. For each visit-linked panel, the fixed-effect design matrix from expanded covariates is evaluated. Max VIF summarizes the maximum column-level VIF after removing the intercept. Rank-deficient indicates exact aliasing in the design matrix.

safe_vif_max <- function(X) {
  if (!is.matrix(X) || nrow(X) < 5 || ncol(X) < 2) return(NA_real_)
  keep <- apply(X, 2, function(col) stats::sd(col, na.rm = TRUE) > 0)
  X <- X[, keep, drop = FALSE]
  if (ncol(X) < 2) return(NA_real_)

  vifs <- sapply(seq_len(ncol(X)), function(j) {
    y <- X[, j]
    Z <- X[, -j, drop = FALSE]
    fit <- tryCatch(lm(y ~ Z), error = function(e) NULL)
    if (is.null(fit)) return(NA_real_)
    r2 <- summary(fit)$r.squared
    if (!is.finite(r2) || r2 >= 1) return(Inf)

3.9 Exposure-covariate screening

Purposeful Selection Screening

Purposeful-style screening checks whether candidate variables show empirical associations with the exposures and/or outcomes, which can support sensitivity-adjustment decisions but does not establish confounding on its own. * A significance threshold of \(p < 0.20\) is typically used to screen potential candidates rather than the standard \(p < 0.05\) to avoid omitting weak confounders. * Small \(q\)-values (BH adjusted) indicate associations that survive multiple-testing adjustments.

This table screens covariate associations with maternal diet exposures across visit-linked clinical panels.

Tests by variable type:

  • Spearman for continuous exposure and continuous covariate pairs.
  • Kruskal-Wallis for continuous exposure vs categorical covariate, and categorical exposure vs continuous covariate.
  • Fisher exact for categorical exposure and categorical covariate pairs.

Tests is the number of exposure-covariate tests in the row. Min p is the smallest raw p-value across those tests. Min q is the smallest BH-adjusted p-value within visit panel. q < 0.10 is the number of exposure-covariate tests with q < 0.10.

var_kind_husna <- function(x, nm = NULL) {
  if (!is.null(nm) && nm %in% categorical_vars_husna) return("categorical")
  if (is.character(x) || is.factor(x)) return("categorical")

assoc_test_husna <- function(dat, exposure, covar, panel_name) {
  if (!all(c(exposure, covar) %in% names(dat))) {
    return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "missing", n = 0L, p.value = NA_real_))
  }
  dd <- dat[, c(exposure, covar), drop = FALSE]
  dd <- dd[stats::complete.cases(dd), , drop = FALSE]
  if (nrow(dd) < 20) {
    return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "too_few_complete", n = nrow(dd), p.value = NA_real_))
  }

  x_kind <- var_kind_husna(dd[[exposure]], exposure)
  z_kind <- var_kind_husna(dd[[covar]], covar)
  x_num <- suppressWarnings(as.numeric(dd[[exposure]]))
  z_num <- suppressWarnings(as.numeric(dd[[covar]]))

  if (x_kind == "continuous" && z_kind == "continuous") {
    tst <- suppressWarnings(cor.test(x_num, z_num, method = "spearman", exact = FALSE))
    return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "Spearman", n = nrow(dd), p.value = tst$p.value))
  }
  if (x_kind == "continuous" && z_kind == "categorical") {
    g <- factor(as.character(dd[[covar]]))
    if (nlevels(g) < 2) return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), p.value = NA_real_))
    tst <- suppressWarnings(kruskal.test(x_num ~ g))
    return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "Kruskal-Wallis", n = nrow(dd), p.value = tst$p.value))
  }
  if (x_kind == "categorical" && z_kind == "continuous") {
    g <- factor(as.character(dd[[exposure]]))
    if (nlevels(g) < 2) return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), p.value = NA_real_))
    tst <- suppressWarnings(kruskal.test(z_num ~ g))
    return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "Kruskal-Wallis", n = nrow(dd), p.value = tst$p.value))
  }

  tab <- table(factor(as.character(dd[[exposure]])), factor(as.character(dd[[covar]])))
  if (any(dim(tab) < 2)) return(tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), p.value = NA_real_))
  tst <- suppressWarnings(fisher.test(tab))
  tibble(panel = panel_name, exposure = exposure, covariate = covar, test = "Fisher exact", n = nrow(dd), p.value = tst$p.value)
}

3.10 Covariate-outcome screening

This table screens covariate associations with metabolite outcome structure. PCA is run on visit-specific metabolite matrices from the SPSS sheet, and each covariate is tested against the first five metabolite PCs within each visit panel.

  • Continuous covariates are screened with Spearman correlation against each PC.
  • Categorical covariates are screened with Kruskal-Wallis tests against each PC.
  • BH correction is applied within visit panel.

make_pc_scores_husna <- function(dat, feature_cols, prefix) {
  feature_cols <- intersect(feature_cols, names(dat))
  if (length(feature_cols) == 0) return(dat %>% select(StudyID))
  mat <- as.matrix(dat[, feature_cols, drop = FALSE])
  mat <- apply(mat, 2, function(x) suppressWarnings(as.numeric(x)))
  if (is.null(dim(mat))) mat <- matrix(mat, ncol = 1L)
  keep <- colMeans(!is.na(mat)) >= 0.7 & apply(mat, 2, stats::sd, na.rm = TRUE) > 0
  mat <- mat[, keep, drop = FALSE]
  if (ncol(mat) == 0) return(dat %>% select(StudyID))
  for (j in seq_len(ncol(mat))) {
    mat[is.na(mat[, j]), j] <- stats::median(mat[, j], na.rm = TRUE)
  }
  pcs <- stats::prcomp(mat, center = TRUE, scale. = TRUE)
  scores <- as.data.frame(pcs$x[, seq_len(min(5L, ncol(pcs$x))), drop = FALSE])
  names(scores) <- paste0(prefix, "_PC", seq_along(scores))
  bind_cols(dat %>% select(StudyID), scores)
}

pc_assoc_husna <- function(dat, covar, pc, panel_name) {
  if (!all(c(covar, pc) %in% names(dat))) {
    return(tibble(panel = panel_name, covariate = covar, outcome_summary = pc, test = "missing", n = 0L, p.value = NA_real_))
  }
  dd <- dat[, c(covar, pc), drop = FALSE]
  dd <- dd[stats::complete.cases(dd), , drop = FALSE]
  if (nrow(dd) < 20) {
    return(tibble(panel = panel_name, covariate = covar, outcome_summary = pc, test = "too_few_complete", n = nrow(dd), p.value = NA_real_))
  }

  y <- suppressWarnings(as.numeric(dd[[pc]]))
  if (var_kind_husna(dd[[covar]], covar) == "continuous") {
    x <- suppressWarnings(as.numeric(dd[[covar]]))
    tst <- suppressWarnings(stats::cor.test(x, y, method = "spearman", exact = FALSE))
    return(tibble(panel = panel_name, covariate = covar, outcome_summary = pc, test = "Spearman", n = nrow(dd), p.value = tst$p.value))
  }
  g <- factor(as.character(dd[[covar]]))
  if (nlevels(g) < 2) {
    return(tibble(panel = panel_name, covariate = covar, outcome_summary = pc, test = "constant", n = nrow(dd), p.value = NA_real_))
  }
  tst <- suppressWarnings(stats::kruskal.test(y ~ g))
  tibble(panel = panel_name, covariate = covar, outcome_summary = pc, test = "Kruskal-Wallis", n = nrow(dd), p.value = tst$p.value)
}

beta_change_ref_husna <- function(df, exposure, covars, candidate) {
  need <- unique(c("Y_ref", exposure, covars))
  need <- need[need %in% names(df)]
  if (!all(c("Y_ref", exposure, candidate) %in% need)) return(c(NA_real_, NA_integer_, NA_integer_))
  d <- df[, need, drop = FALSE]
  d <- d[stats::complete.cases(d), , drop = FALSE]
  if (nrow(d) < 30) return(c(NA_real_, nrow(d), NA_integer_))
  if (!candidate %in% covars || !candidate %in% names(d)) return(c(NA_real_, nrow(d), NA_integer_))

  cov_full <- covars[covars %in% names(d)]
  cov_red <- setdiff(cov_full, candidate)
  f_full <- as.formula(paste("Y_ref ~", paste(c(exposure, cov_full), collapse = " + ")))
  f_red <- as.formula(paste("Y_ref ~", paste(c(exposure, cov_red), collapse = " + ")))
  m_full <- tryCatch(lm(f_full, data = d), error = function(e) NULL)
  m_red <- tryCatch(lm(f_red, data = d), error = function(e) NULL)
  if (is.null(m_full) || is.null(m_red)) return(c(NA_real_, nrow(d), NA_integer_))
  b_full <- coef(m_full)[exposure]
  b_red <- coef(m_red)[exposure]
  if (is.na(b_full) || is.na(b_red) || !is.finite(b_full) || abs(b_full) < 1e-8) return(c(NA_real_, nrow(d), nrow(model.frame(m_red))))
  c(abs((b_red - b_full) / b_full) * 100, nrow(model.frame(m_full)), nrow(model.frame(m_red)))
}

4 Model specification

4.1 Proposed models

  1. Age-specific linear model: Y_age ~ E + maternal_core_covariates + visit_specific_fasting
  2. Mutually adjusted timing model: Y_age ~ E_early + E_late + maternal_core_covariates + visit_specific_fasting
  3. Repeated-measures mixed model: Y_it ~ E + age_t + E:age_t + maternal_core_covariates + (1|child)
  4. Sensitivity models: add sensitivity covariate block and compare estimates.

4.2 Mathematical model forms

  1. Age-specific linear model (age a):

\[ Y_{i,a} = \beta_0 + \beta_1 E_i + \sum_{k=1}^{p} \gamma_k C_{i,k} + \delta F_{i,a} + \varepsilon_{i,a} \]

  1. Mutually adjusted timing model (age a):

\[ Y_{i,a} = \beta_0 + \beta_1 E_{i,\mathrm{early}} + \beta_2 E_{i,\mathrm{late}} + \sum_{k=1}^{p} \gamma_k C_{i,k} + \delta F_{i,a} + \varepsilon_{i,a} \]

  1. Repeated-measures mixed model (child i, time t):

\[ Y_{i,t} = \beta_0 + \beta_1 E_i + \beta_2 t + \beta_3(E_i \times t) + \sum_{k=1}^{p} \gamma_k C_{i,k} + u_i + \varepsilon_{i,t} \]

with random intercept:

\[ u_i \sim \mathcal{N}(0, \sigma_u^2), \quad \varepsilon_{i,t} \sim \mathcal{N}(0, \sigma^2) \]

build_variable_key <- function(df) {
  nm <- names(df)
  make_friendly <- function(x) {
    y <- gsub("([a-z])([A-Z])", "\\1 \\2", x)
    y <- gsub("([A-Za-z])([0-9])", "\\1 \\2", y)
    y <- gsub("([0-9])([A-Za-z])", "\\1 \\2", y)
    y <- gsub("[._]+", " ", y)
    y <- gsub("\\s+", " ", trimws(y))
    tools::toTitleCase(tolower(y))
  }
  var_label <- sapply(nm, function(v) {
    lb <- attr(df[[v]], "label")
    if (is.null(lb) || is.na(lb) || lb == "") "" else as.character(lb)
  })
  value_levels <- sapply(nm, function(v) {
    labs <- attr(df[[v]], "labels")
    if (is.null(labs) || length(labs) == 0) return("")
    paste(sprintf("%s = %s", names(labs), unname(labs)), collapse = "; ")
  })
  data.frame(
    Friendly_Name = sapply(nm, make_friendly),
    Variable_Code = nm,
    Variable_Label = var_label,
    Value_Levels = value_levels,
    stringsAsFactors = FALSE
  )
}

4 Supplementary Table S4. Returned result inventory

Show code 
q4_main <- q4 %>% filter(!grepl(":", term))
q4_interaction <- q4 %>% filter(grepl(":", term))

inventory_tbl <- bind_rows(
  add_exposure_family(q1) %>% mutate(component = "Concurrent at 2 Years", n_col = n),
  add_exposure_family(q2) %>% mutate(component = "Concurrent at 5-6 Years", n_col = n),
  add_exposure_family(q3) %>% mutate(component = "Prospective: 2-Year Diet to 5-6-Year Metabolites", n_col = n),
  add_exposure_family(q4_main) %>% mutate(component = "Repeated-Measures Main Effect", n_col = n_obs),
  add_exposure_family(q4_interaction) %>% mutate(component = "Repeated-Measures Diet-by-Time Interaction", n_col = n_obs)
) %>%
  group_by(component, exposure_family) %>%
  summarise(
    result_rows = n(),
    significant_rows_q_lt_0_10 = sum(q.value < 0.10, na.rm = TRUE),
    complete_case_n_range = {
      vals <- sort(unique(n_col[is.finite(n_col)]))
      if (length(vals) == 0) NA_character_ else paste0(min(vals), "-", max(vals))
    },
    .groups = "drop"
  ) %>%
  mutate(exposure_family = factor(exposure_family, levels = family_order)) %>%
  arrange(component, exposure_family)

make_table(inventory_tbl, caption = "Supplementary Table S4. Returned result inventory")

5 Supplementary Table S5. Candidate associations with q < 0.10

Show code 
candidate_linear <- bind_rows(
  q1 %>% mutate(component = "Concurrent at 2 Years"),
  q2 %>% mutate(component = "Concurrent at 5-6 Years"),
  q3 %>% mutate(component = "Prospective: 2-Year Diet to 5-6-Year Metabolites")
) %>%
  filter(q.value < 0.10) %>%
  mutate(multilevel_factor = grepl("\\|", term)) %>%
  transmute(
    component,
    exposure_family = exposure_family(exposure),
    exposure = pretty_var(exposure),
    feature,
    term = if_else(multilevel_factor, "Omnibus multi-level factor test", NA_character_),
    n,
    beta = if_else(multilevel_factor, NA_real_, estimate),
    conf_low = if_else(multilevel_factor, NA_real_, conf.low),
    conf_high = if_else(multilevel_factor, NA_real_, conf.high),
    p_value = p.value,
    q_value = q.value
  )

candidate_q4 <- bind_rows(
  q4_main %>% mutate(component = "Repeated-Measures Main Effect"),
  q4_interaction %>% mutate(component = "Repeated-Measures Diet-by-Time Interaction")
) %>%
  filter(q.value < 0.10) %>%
  transmute(
    component,
    exposure_family = exposure_family(exposure),
    exposure = pretty_var(exposure),
    feature,
    term,
    n_obs,
    n_ind,
    beta = estimate,
    conf_low = estimate - 1.96 * std.error,
    conf_high = estimate + 1.96 * std.error,
    p_value = p.value,
    q_value = q.value
  )

candidate_tbl <- bind_rows(
  candidate_linear %>% mutate(n_obs = n, n_ind = NA_real_),
  candidate_q4 %>% rename(n = n_obs)
) %>%
  mutate(exposure_family = factor(exposure_family, levels = family_order)) %>%
  arrange(exposure_family, q_value, p_value)

make_table(candidate_tbl, caption = "Supplementary Table S5. Candidate associations meeting q < 0.10")

stability_tbl <- bind_rows(
  compare_primary_sensitivity(q1, q1_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Concurrent at 2 Years"),
  compare_primary_sensitivity(q2, q2_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Concurrent at 5-6 Years"),
  compare_primary_sensitivity(q3, q3_sens, c("exposure", "feature", "visit_exposure", "visit_outcome")) %>% mutate(component = "Prospective: 2-Year Diet to 5-6-Year Metabolites"),
  compare_primary_sensitivity(q4, q4_sens, c("exposure", "feature", "term")) %>% mutate(component = "Repeated-Measures")
) %>%
  mutate(exposure_family = exposure_family(exposure)) %>%
  group_by(component, exposure_family) %>%
  summarise(
    matched_rows = n(),
    median_abs_beta_change = median(abs_delta, na.rm = TRUE),
    p90_abs_beta_change = quantile(abs_delta, 0.90, na.rm = TRUE),
    direction_flip_fraction = mean(direction_flip, na.rm = TRUE),
    q_reclassification_fraction = mean(q_reclassified, na.rm = TRUE),
    .groups = "drop"
  )

make_table(stability_tbl, caption = "Supplementary Table S6. Primary-vs-sensitivity stability summary")

status_tbl <- model_status %>%
  mutate(
    component = case_when(
      question == "Q1" ~ "Concurrent at 2 Years",
      question == "Q2" ~ "Concurrent at 5-6 Years",
      question == "Q3" ~ "Prospective: 2-Year Diet to 5-6-Year Metabolites",
      question == "Q4" ~ "Repeated-Measures",
      TRUE ~ question
    ),
    exposure = pretty_var(exposure),
    complete_case_n = ifelse(
      is.na(n_complete_min) | !is.finite(n_complete_min),
      NA_character_,
      ifelse(n_complete_min == n_complete_max, as.character(n_complete_min), paste0(n_complete_min, "-", n_complete_max))
    )
  ) %>%
  select(component, analysis, exposure, term_family, status, reason, complete_case_n, n_features_tested)

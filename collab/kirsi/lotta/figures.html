burden_tbl <- bind_rows(

4 Inferential signal summary

Show code 
signal_tbl <- bind_rows(
  add_exposure_family(q1) %>% mutate(component = "Concurrent at 2 Years"),
  add_exposure_family(q2) %>% mutate(component = "Concurrent at 5-6 Years"),
  add_exposure_family(q3) %>% mutate(component = "Prospective"),
  add_exposure_family(q4) %>% mutate(component = ifelse(grepl(":", term), "Repeated-Measures: interactions", "Repeated-Measures: main effects"))
) %>%
  group_by(component, exposure_family) %>%
  summarise(significant_rows = sum(q.value < 0.1, na.rm = TRUE), total_rows = n(), .groups = "drop") %>%
  mutate(
    exposure_family = factor(exposure_family, levels = family_order),
    significant_fraction = significant_rows / total_rows
  )

signal_tbl %>%
  ggplot(aes(x = component, y = significant_fraction, fill = exposure_family)) +
  geom_col(width = 0.65, position = position_dodge(width = 0.75)) +
  geom_text(
    aes(label = paste0(scales::percent(significant_fraction, accuracy = 0.1), "\n(", significant_rows, "/", total_rows, ")")),
    position = position_dodge(width = 0.75),
    vjust = -0.2,
    size = 3.3
  ) +
  labs(x = "Component", y = "Fraction of rows with q < 0.10", fill = "Exposure family") +
  theme_minimal(base_size = 15) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

5 Primary-vs-sensitivity stability

Show code 
comp_tbl <- bind_rows(
  compare_primary_sensitivity(q1, q1_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Concurrent at 2 Years"),
  compare_primary_sensitivity(q2, q2_sens, c("exposure", "feature", "visit")) %>% mutate(component = "Concurrent at 5-6 Years"),
  compare_primary_sensitivity(q3, q3_sens, c("exposure", "feature", "visit_exposure", "visit_outcome")) %>% mutate(component = "Prospective"),
  compare_primary_sensitivity(q4, q4_sens, c("exposure", "feature", "term")) %>% mutate(component = "Repeated-Measures")
)

stability_tbl <- comp_tbl %>%
  mutate(exposure_family = exposure_family(exposure)) %>%
  group_by(component, exposure_family) %>%
  summarise(
    pairs_compared = n(),
    median_abs_delta = median(abs_delta, na.rm = TRUE),
    q_reclassification_fraction = mean(q_reclassified, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    median_abs_delta = round(median_abs_delta, 3),
    q_reclassification_fraction = round(q_reclassification_fraction, 3)
  )

stability_tbl %>%
  pivot_longer(cols = c(median_abs_delta, q_reclassification_fraction), names_to = "metric", values_to = "value") %>%
  mutate(metric = recode(metric,
    median_abs_delta = "Median absolute beta change",
    q_reclassification_fraction = "Fraction reclassified at q < 0.10"
  )) %>%
  mutate(exposure_family = factor(exposure_family, levels = family_order)) %>%
  ggplot(aes(x = component, y = value, fill = exposure_family)) +
  geom_col(width = 0.65, position = position_dodge(width = 0.75)) +
  facet_wrap(~ metric, scales = "free_y") +
  labs(x = "Component", y = "Value", fill = "Exposure family") +
  theme_minimal(base_size = 15) +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

rep_q1 <- pick_representative_exposure(q1)
rep_q2 <- pick_representative_exposure(q2)
rep_q3 <- pick_representative_exposure(q3)

volcano_df <- bind_rows(
  q1 %>% filter(exposure == rep_q1) %>% mutate(component = "Concurrent at 2 Years", sig = q.value < 0.1),
  q2 %>% filter(exposure == rep_q2) %>% mutate(component = "Concurrent at 5-6 Years", sig = q.value < 0.1),
  q3 %>% filter(exposure == rep_q3) %>% mutate(component = "Prospective", sig = q.value < 0.1)
) %>%
  mutate(logp = safe_logp(p.value))

volcano_df %>%
  ggplot(aes(x = estimate, y = logp, color = sig)) +
  geom_point(alpha = 0.65, size = 1.6) +
  geom_vline(xintercept = 0, linetype = "dashed", color = "grey40") +
  facet_wrap(~ component, scales = "free") +
  scale_color_manual(values = c("FALSE" = "grey65", "TRUE" = "firebrick")) +
  labs(x = "Beta", y = "-log10(p)", color = "q < 0.10") +
  theme_minimal(base_size = 16)

9 Prospective overlap and concordance

Show code 
sig_q1 <- unique(q1$feature[q1$q.value < 0.1])
sig_q2 <- unique(q2$feature[q2$q.value < 0.1])
sig_q3 <- unique(q3$feature[q3$q.value < 0.1])
all_feat <- unique(c(sig_q1, sig_q2, sig_q3))

ov <- data.frame(feature = all_feat, stringsAsFactors = FALSE) %>%
  mutate(
    q1 = feature %in% sig_q1,
    q2 = feature %in% sig_q2,
    q3 = feature %in% sig_q3,
    signature = paste0(ifelse(q1, "2y", "-"), " | ", ifelse(q2, "5-6y", "-"), " | ", ifelse(q3, "Prospective", "-"))
  ) %>%
  count(signature, name = "n_features") %>%
  arrange(desc(n_features))

ov %>%
  ggplot(aes(x = fct_reorder(signature, n_features), y = n_features, fill = signature)) +
  geom_col(show.legend = FALSE) +
  coord_flip() +
  labs(x = "Signature membership", y = "Metabolite count") +
  theme_minimal(base_size = 15)

visit_label <- function(v) dplyr::case_when(
  v == 6L ~ "2 Years",
  v == 7L ~ "5-6 Years",
  TRUE ~ paste("Visit", v)
)

pretty_var <- function(x) {
  x <- gsub("CEPercent", "Percent ", x)
  x <- gsub("CFibtot", "Fibre ", x)
  x <- gsub("C_", "", x)
  x <- gsub("DietQuality", "Diet Quality ", x)
  x <- gsub("CHO", "Carbohydrate", x)
  x <- gsub("Pro", "Protein", x)
  x <- gsub("FAn3", "n-3 fatty acids", x)
  x <- gsub("FAn6", "n-6 fatty acids", x)
  x <- gsub("SFA", "SFA", x)
  x <- gsub("MUFA", "MUFA", x)
  x <- gsub("PUFA", "PUFA", x)
  x <- gsub("Diet6", "Diet", x)
  x <- gsub("Diet7", "Diet", x)
  x <- gsub("_6", "", x, fixed = TRUE)
  x <- gsub("_7", "", x, fixed = TRUE)
  x <- gsub("_", " ", x, fixed = TRUE)
  x <- gsub("\\s+", " ", trimws(x))
  tools::toTitleCase(tolower(x))
}

exposure_vars <- c(
  "CEPercentProDiet6", "CEPercentCHODiet6", "CEPercentFatDiet6", "CEPercentSFADiet6",
  "CEPercentMUFADiet6", "CEPercentPUFADiet6", "CEPercentFAn3Diet6", "CEPercentFAn6Diet6",
  "CFibtotDiet6", "C_DietQuality_points_6", "C_DietQuality_classes_6",
  "CEPercentProDiet7", "CEPercentCHODiet7", "CEPercentFatDiet7", "CEPercentSFADiet7",
  "CEPercentMUFADiet7", "CEPercentPUFADiet7", "CEPercentFAn3Diet7", "CEPercentFAn6Diet7",
  "CFibtotDiet7", "C_DietQuality_points_7", "C_DietQuality_classes_7"
)

exposure_tbl <- bind_rows(lapply(sort(unique(df$visit_num)), function(v) {
  dv <- df %>% filter(visit_num == v)

  current_exposures <- if (v == 6L) {
    exposure_vars[grepl("6$|_6$", exposure_vars)]
  } else {
    exposure_vars[grepl("7$|_7$", exposure_vars)]
  }

  bind_rows(lapply(current_exposures, function(var_nm) {
    x <- dv[[var_nm]]
    x_num <- suppressWarnings(as.numeric(as.character(x)))
    is_continuous <- sum(is.finite(x_num)) >= 20L && length(unique(x_num[is.finite(x_num)])) >= 8L

    if (is_continuous) {
      data.frame(
        child_age = visit_label(v),
        exposure = pretty_var(var_nm),
        variable = var_nm,
        type = "continuous",
        non_missing = sum(!is.na(x_num)),
        summary = paste0(
          round(mean(x_num, na.rm = TRUE), 3),
          " (sd ", round(sd(x_num, na.rm = TRUE), 3), "); median ",
          round(stats::median(x_num, na.rm = TRUE), 3)
        ),
        stringsAsFactors = FALSE
      )
    } else {
      lev <- sort(table(as.character(x)), decreasing = TRUE)
      top_level <- names(lev)[1]
      top_n <- if (length(lev) == 0) 0 else unname(lev[1])
      data.frame(
        child_age = visit_label(v),
        exposure = pretty_var(var_nm),
        variable = var_nm,
        type = "categorical",
        non_missing = sum(!is.na(x)),
        summary = paste0("most common: ", top_level, " (", top_n, ")"),
        stringsAsFactors = FALSE
      )
    }
  }))
}))
make_table(exposure_tbl, caption = "Exposure availability and distributions by child age")

metabolite_missing <- bind_rows(lapply(sort(unique(df$visit_num)), function(v) {
  dv <- df %>% filter(visit_num == v)
  miss <- colMeans(is.na(dv[, metabolite_cols, drop = FALSE]))

    child_age = visit_label(v),

    median_missing_fraction = median(miss),

  )
}))

6 Global structure (PCA) with top loadings

Show code 
pca_details <- compute_pca_details(
  df = df,
  feature_cols = metabolite_cols,
  group_cols = c("visit"),
  min_feature_obs = 0.7,
  top_n_loadings = 5L
)

if (is.null(pca_details)) {
  ggplot() +
    annotate("text", x = 0, y = 0, label = "PCA unavailable: insufficient complete feature matrix") +
    theme_void()
} else {
  scores <- pca_details$scores
  scores$visit <- factor(gsub("_", " ", scores$visit), levels = sort(unique(gsub("_", " ", scores$visit))))
  visit_counts <- scores %>%
    count(visit, name = "n") %>%
    mutate(label = paste0(visit, " (n=", n, ")"))
  label_map <- setNames(visit_counts$label, as.character(visit_counts$visit))
  legend_labels <- unname(label_map[levels(scores$visit)])
  top_loadings_plot <- pca_details$top_loadings
  score_scale <- max(abs(scores$PC1), abs(scores$PC2), na.rm = TRUE)

  has_ellipse <- scores %>%
    count(visit, name = "n") %>%
    summarise(ok = all(n >= 3) & n() >= 2) %>%
    pull(ok)

  p <- ggplot(scores, aes(x = PC1, y = PC2, color = visit)) +
    geom_point(alpha = 0.7, size = 1.7) +
    theme_minimal(base_size = 11) +
    coord_equal() +
    scale_color_brewer(palette = "Set1", labels = legend_labels, drop = FALSE) +
    labs(
      title = "PCA across child ages",
      x = paste0("PC1 (", round(100 * pca_details$var_explained[1], 1), "%)"),
      y = paste0("PC2 (", round(100 * pca_details$var_explained[2], 1), "%)"),
      color = "Visit"
    ) +
    theme(
      plot.title = element_text(hjust = 0.5),
      legend.position = "top",
      panel.grid.minor = element_blank()
    )

  if (isTRUE(has_ellipse)) {
    p <- p + stat_ellipse(linewidth = 0.8, alpha = 0.7)
  }

  p +
    geom_segment(
      data = top_loadings_plot,
      aes(x = 0, y = 0, xend = PC1_scaled, yend = PC2_scaled),
      inherit.aes = FALSE,
      color = "black",
      linewidth = 0.45,
      arrow = arrow(length = grid::unit(0.18, "cm"))
    ) +
    geom_text(
      data = top_loadings_plot,
      aes(x = PC1_scaled, y = PC2_scaled, label = feature),
      inherit.aes = FALSE,
      size = 2.8,
      color = "black",
      check_overlap = TRUE,
      nudge_y = 0.02 * score_scale
    )
}

8 Clinical snapshot

Show code 
snapshot_vars <- c("Intervention", "MAge1", "MprepBMI", "MPrimipara", "MUniEdu", "MPRSmoke", "MGDMOGTT1OR2Fi")
snapshot_vars <- snapshot_vars[snapshot_vars %in% names(df)]

if (length(snapshot_vars) == 0) {
  snapshot_tbl <- data.frame(message = "Clinical snapshot unavailable")
} else {
  snapshot_tbl <- bind_rows(lapply(sort(unique(df$visit_num)), function(v) {
    dv <- df %>% filter(visit_num == v)

    bind_rows(lapply(snapshot_vars, function(var_nm) {
      x <- dv[[var_nm]]
      x_num <- suppressWarnings(as.numeric(as.character(x)))

      if (all(!is.na(x_num))) {
        data.frame(
          child_age = visit_label(v),
          variable = pretty_var(var_nm),
          variable_code = var_nm,
          type = "numeric",
          n_non_missing = sum(!is.na(x_num)),
          summary = paste0(round(mean(x_num, na.rm = TRUE), 2), " (sd ", round(sd(x_num, na.rm = TRUE), 2), ")"),
          stringsAsFactors = FALSE
        )
      } else {
        top_level <- names(sort(table(x), decreasing = TRUE))[1]
        top_n <- max(table(x), na.rm = TRUE)
        data.frame(
          child_age = visit_label(v),
          variable = pretty_var(var_nm),
          variable_code = var_nm,
          type = "categorical",
          n_non_missing = sum(!is.na(x)),
          summary = paste0("top: ", top_level, " (", top_n, ")"),
          stringsAsFactors = FALSE
        )
      }
    }))
  }))
}

make_table(snapshot_tbl, caption = "Compact clinical snapshot by child age")

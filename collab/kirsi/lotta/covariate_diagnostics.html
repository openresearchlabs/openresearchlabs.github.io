show_table <- function(tbl, page_len = 10, caption = NULL, title = NULL, colnames = NULL) {
  if (!has_DT) return(tbl)
  
  tbl_dt <- DT::datatable(
    tbl,
    caption = caption,
    rownames = FALSE,
    filter = "top",
    escape = FALSE,
    colnames = if (is.null(colnames)) names(tbl) else colnames,
    options = list(
      pageLength = page_len,
      scrollX = FALSE,
      autoWidth = FALSE
    )
  )

  # Conditional formatting based on column name presence
  # 1. Suggested status (green/yellow/gray)
  if ("Suggested status" %in% names(tbl)) {
    tbl_dt <- tbl_dt %>% DT::formatStyle(
      "Suggested status",
      target = "row",
      backgroundColor = DT::styleEqual(
        c("keep_core", "keep_sensitivity", "consider_drop"),
        c("#e2f0d9", "#fff2cc", "#f2f2f2")
      )
    )
  }

  # 2. Collinearity flagged/flag (soft red background for whole row)
  if ("Collinearity flag" %in% names(tbl)) {
    tbl_dt <- tbl_dt %>% DT::formatStyle(
      "Collinearity flag",
      target = "row",
      backgroundColor = DT::styleEqual("yes", "#fce4e4")
    )
  }
  if ("flagged" %in% names(tbl)) {
    tbl_dt <- tbl_dt %>% DT::formatStyle(
      "flagged",
      target = "row",
      backgroundColor = DT::styleEqual(TRUE, "#fce4e4")
    )
  }

  # 3. High missingness (Missing_pct): soft orange (>10%), soft red (>20%)
  if ("Missing_pct" %in% names(tbl)) {
    tbl_dt <- tbl_dt %>% DT::formatStyle(
      "Missing_pct",
      backgroundColor = DT::styleInterval(c(10, 20), c("transparent", "#ffe6cc", "#fce4e4"))
    )
  }

  if (!is.null(title) && nzchar(title)) {
    return(htmltools::tagList(htmltools::tags$h4(title), tbl_dt))
  }

  tbl_dt
}

3 Covariate specification

3.1 Covariate selection framework

Covariate selection is pre-specified and DAG-informed:

  1. The primary adjustment set is fixed in advance for each visit.
  2. Balance tests and missingness summaries are reported.
  3. Collinearity diagnostics are reported to flag potentially unstable expanded adjustment sets.
  4. Secondary models add sensitivity covariates.

3.2 Group balance tests across intervention arms

Why this matters: Balance Diagnostics

These tests check whether covariates are balanced across the intervention arms. Because this study evaluates a randomized trial context, minor imbalances can arise by chance. Note that standard guidelines discourage using p-values alone to drop covariates—pre-specified confounders are retained regardless of balance test significance.

Covariate balance diagnostics across intervention groups:

  • Kruskal-Wallis for continuous covariates.
  • Fisher exact for discrete covariates (ordinal and nominal categorical).
  • BH adjusted p-values across all visits in the combined table.

These are covariate-intervention association tests used as balance diagnostics. They are not outcome-association tests, and their p-values are not used alone to choose adjustment covariates. The intervention variable itself is the grouping variable and is therefore not included as a balance-test covariate.

run_covariate_test <- function(df, group_var, covariate, test_method) {
  out <- data.frame(
    var = covariate,
    test = test_method,
    p_value = NA_real_,
    p_adj_bh = NA_real_,
    n_groups = NA_integer_,
    note = NA_character_,
    stringsAsFactors = FALSE
  )

  if (!covariate %in% names(df)) {
    out$note <- "covariate_missing_in_data"
    return(out)
  }
  if (!group_var %in% names(df)) {
    out$note <- "group_var_missing_in_data"
    return(out)
  }
  if (!test_method %in% c("anova", "kruskal_wallis", "fisher_or_chisq")) {
    out$note <- "invalid_test_method"
    return(out)
  }

  x <- df[[covariate]]
  g <- df[[group_var]]
  keep <- !is.na(x) & !is.na(g)
  if (!any(keep)) {
    out$note <- "no_complete_cases"
    return(out)
  }

  x <- x[keep]
  g <- as.factor(g[keep])
  out$n_groups <- nlevels(g)

  if (out$n_groups < 2L) {
    out$note <- "fewer_than_two_groups"
    return(out)
  }
  if (length(unique(x)) < 2L) {
    out$note <- "constant_covariate"
    return(out)
  }

  if (identical(test_method, "anova")) {
    out$p_value <- tryCatch({
      fit <- aov(as.numeric(x) ~ g)
      summary(fit)[[1]][["Pr(>F)"]][1]
    }, error = function(e) NA_real_)
  } else if (identical(test_method, "kruskal_wallis")) {
    out$test <- "kruskal_wallis"
    y <- if (is.character(x) || is.factor(x)) as.numeric(factor(x)) else as.numeric(x)
    out$p_value <- tryCatch(kruskal.test(y ~ g)$p.value, error = function(e) NA_real_)
  } else {
    x_cat <- as.factor(x)
    if (nlevels(x_cat) < 2L) {
      out$note <- "fewer_than_two_levels"
      return(out)
    }

    tbl <- table(g, x_cat)
    if (any(dim(tbl) < 2L)) {
      out$note <- "invalid_contingency_table"
      return(out)
    }

    out$p_value <- tryCatch({
      out$test <- "fisher_exact"
      fisher.test(tbl)$p.value
    }, error = function(e) NA_real_)
  }

  out$note <- ifelse(is.na(out$p_value), "test_failed", "ok")
  out
}

run_group_balance_panel <- function(df, group_var, covariates, panel_name) {
  res <- bind_rows(lapply(seq_len(nrow(covariates)), function(i) {
    run_covariate_test(
      df = df,
      group_var = group_var,
      covariate = covariates$var[i],
      test_method = covariates$test[i]
    )
  }))
  res$panel <- panel_name
  res %>% arrange(p_value, .by_group = FALSE)
}

3.8 Collinearity

Understanding Collinearity & VIF
  • Pairwise Screen: A Spearman correlation or Cramer’s V \(\ge 0.70\) highlights redundant covariate pairs. In such cases, standard errors of the exposure effect estimate may inflate if both are kept.
  • Model-Matrix VIF: Variance Inflation Factors (VIF) summarize design-wide multicollinearity. Values below 5 indicate no concern. Values above 5 (or rank deficiency indicating exact aliasing) require reviewing the models for redundant parameters or category sparseness.

Collinearity is checked before interpreting expanded adjustment sets:

  • Numeric covariate pairs are screened with Spearman correlations within each visit-specific analytic panel.
  • Categorical covariate pairs are screened with Cramer’s V.
  • Pairs with absolute Spearman correlation or Cramer’s V at or above 0.70 are flagged for review.

cramers_v <- function(x, y) {
  keep <- !is.na(x) & !is.na(y)
  x <- as.factor(x[keep])
  y <- as.factor(y[keep])
  if (length(x) == 0 || nlevels(x) < 2L || nlevels(y) < 2L) return(NA_real_)
  tab <- table(x, y)
  chi <- suppressWarnings(chisq.test(tab, correct = FALSE))
  n <- sum(tab)
  denom <- n * (min(dim(tab)) - 1)
  if (!is.finite(denom) || denom <= 0) return(NA_real_)
  sqrt(as.numeric(chi$statistic) / denom)
}

collinearity_screen <- function(df, vars, panel_name, threshold = 0.70) {
  vars <- intersect(vars, names(df))
  vars <- vars[vapply(vars, function(v) sum(!is.na(df[[v]])) >= 10, logical(1))]
  if (length(vars) < 2L) return(data.frame())

  pairs <- utils::combn(vars, 2, simplify = FALSE)
  bind_rows(lapply(pairs, function(pair) {
    x <- df[[pair[[1]]]]
    y <- df[[pair[[2]]]]
    x_num <- is.numeric(x) || inherits(x, "labelled")
    y_num <- is.numeric(y) || inherits(y, "labelled")

    if (x_num && y_num) {
      value <- suppressWarnings(cor(as.numeric(x), as.numeric(y), use = "pairwise.complete.obs", method = "spearman"))
      metric <- "spearman_rho"
      flag <- is.finite(value) && abs(value) >= threshold
    } else {
      value <- cramers_v(x, y)
      metric <- "cramers_v"
      flag <- is.finite(value) && value >= threshold
    }

    data.frame(
      panel = panel_name,
      covariate_1 = pair[[1]],
      covariate_2 = pair[[2]],
      metric = metric,
      value = value,
      flagged = flag,
      stringsAsFactors = FALSE
    )
  })) %>%
    filter(is.finite(value)) %>%
    arrange(desc(abs(value)))
}

3.9 Model-matrix collinearity

Model-matrix collinearity is checked before interpreting expanded adjustment sets. The fixed-effect design matrix is evaluated and the maximum design-column VIF is reported within each term; rank-deficient models indicate exact collinearity or aliasing.

design_vif <- function(dat, exposure, covars, label) {
  rhs <- intersect(c(exposure, covars), names(dat))
  if (length(rhs) < 2 || !exposure %in% rhs) {
    return(tibble(model_family = label, exposure = exposure, term = NA_character_, n = 0L, matrix_rank = NA_integer_, matrix_cols = NA_integer_, rank_deficient = NA, max_column_vif = NA_real_))
  }

  dd <- prep_model_data(dat, rhs)
  dd <- dd[stats::complete.cases(dd), , drop = FALSE]
  if (nrow(dd) < 30) {
    return(tibble(model_family = label, exposure = exposure, term = NA_character_, n = nrow(dd), matrix_rank = NA_integer_, matrix_cols = NA_integer_, rank_deficient = NA, max_column_vif = NA_real_))
  }

  X <- tryCatch(model.matrix(reformulate(rhs), data = dd), error = function(e) NULL)
  if (is.null(X)) {
    return(tibble(model_family = label, exposure = exposure, term = NA_character_, n = nrow(dd), matrix_rank = NA_integer_, matrix_cols = NA_integer_, rank_deficient = NA, max_column_vif = NA_real_))
  }

  assign_vec <- attr(X, "assign")
  term_labels <- attr(terms(reformulate(rhs)), "term.labels")
  X <- X[, colnames(X) != "(Intercept)", drop = FALSE]
  assign_vec <- assign_vec[colnames(model.matrix(reformulate(rhs), data = dd)) != "(Intercept)"]
  keep <- apply(X, 2, function(z) stats::sd(z, na.rm = TRUE) > 0)
  X <- X[, keep, drop = FALSE]
  assign_vec <- assign_vec[keep]

  rank_x <- qr(X)$rank
  p_x <- ncol(X)
  if (p_x < 2) {
    vif_vals <- rep(NA_real_, p_x)
  } else {
    vif_vals <- vapply(seq_len(p_x), function(j) {
      xj <- X[, j]
      others <- X[, -j, drop = FALSE]
      r2 <- tryCatch(summary(lm(xj ~ others))$r.squared, error = function(e) NA_real_)
      if (!is.finite(r2) || r2 >= 1) return(Inf)
      1 / (1 - r2)
    }, numeric(1))
  }

  tibble(
    model_family = label,
    exposure = exposure,
    term = term_labels[assign_vec],
    n = nrow(dd),
    matrix_rank = rank_x,
    matrix_cols = p_x,
    rank_deficient = rank_x < p_x,
    column_vif = vif_vals
  ) %>%
    group_by(model_family, exposure, term, n, matrix_rank, matrix_cols, rank_deficient) %>%
    summarise(max_column_vif = max(column_vif, na.rm = TRUE), .groups = "drop")
}

3.10 Exposure-covariate screening

Purposeful Selection Screening

Purposeful-style screening checks whether candidate variables show empirical associations with the exposures and/or outcomes, which can support sensitivity-adjustment decisions but does not establish confounding on its own. * A significance threshold of \(p < 0.20\) is typically used to screen potential candidates rather than the standard \(p < 0.05\) to avoid omitting weak confounders. * Small \(q\)-values (BH adjusted) indicate associations that survive multiple-testing adjustments.

This table screens covariate associations with diet exposures. Within each visit panel, each covariate is tested against each relevant diet exposure, then summarized as one row per covariate.

Tests by variable type:

  • Spearman for continuous exposure and continuous covariate pairs.
  • Kruskal-Wallis for continuous exposure vs categorical covariate, and categorical exposure vs continuous covariate.
  • Fisher exact for categorical exposure and categorical covariate pairs.

Tests is the number of exposure-covariate tests in the row. Min p is the smallest raw p-value across those tests. Min q is the smallest BH-adjusted p-value within the visit panel. q < 0.10 is the number of exposure-covariate tests for that covariate with q < 0.10.

Interpretation: a covariate with q < 0.10 greater than zero is associated with at least one diet exposure after BH correction at the 10% FDR level. A value of zero means no exposure-covariate association met that threshold in the panel.

assoc_test <- function(dat, exposure, covar, family) {
  if (!all(c(exposure, covar) %in% names(dat))) {
    return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "missing", n = 0L, statistic = NA_real_, p.value = NA_real_))
  }
  dd <- dat[, c(exposure, covar), drop = FALSE]
  dd <- dd[stats::complete.cases(dd), , drop = FALSE]
  if (nrow(dd) < 20) {
    return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "too_few_complete", n = nrow(dd), statistic = NA_real_, p.value = NA_real_))
  }
  x_kind <- var_kind(dd[[exposure]], exposure)
  z_kind <- var_kind(dd[[covar]], covar)
  x_num <- suppressWarnings(as.numeric(dd[[exposure]]))
  z_num <- suppressWarnings(as.numeric(dd[[covar]]))

  if (x_kind == "continuous" && z_kind == "continuous") {
    tst <- suppressWarnings(cor.test(x_num, z_num, method = "spearman", exact = FALSE))
    return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "Spearman", n = nrow(dd), statistic = unname(tst$estimate), p.value = tst$p.value))
  }

  if (x_kind == "continuous" && z_kind == "categorical") {
    g <- factor(as.character(dd[[covar]]))
    if (nlevels(g) < 2) return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), statistic = NA_real_, p.value = NA_real_))
    tst <- suppressWarnings(kruskal.test(x_num ~ g))
    return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "Kruskal-Wallis", n = nrow(dd), statistic = unname(tst$statistic), p.value = tst$p.value))
  }

  if (x_kind == "categorical" && z_kind == "continuous") {
    g <- factor(as.character(dd[[exposure]]))
    if (nlevels(g) < 2) return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), statistic = NA_real_, p.value = NA_real_))
    tst <- suppressWarnings(kruskal.test(z_num ~ g))
    return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "Kruskal-Wallis", n = nrow(dd), statistic = unname(tst$statistic), p.value = tst$p.value))
  }

  tab <- table(factor(as.character(dd[[exposure]])), factor(as.character(dd[[covar]])))
  if (any(dim(tab) < 2)) return(tibble(model_family = family, exposure = exposure, covariate = covar, test = "constant", n = nrow(dd), statistic = NA_real_, p.value = NA_real_))
  tst <- suppressWarnings(fisher.test(tab))
  tibble(model_family = family, exposure = exposure, covariate = covar, test = "Fisher exact", n = nrow(dd), statistic = NA_real_, p.value = tst$p.value)
}

make_pc_scores <- function(dat, features, prefix) {
  features <- intersect(features, names(dat))
  if (length(features) == 0) return(dat %>% select(StudyID))
  mat <- as.matrix(dat[, features, drop = FALSE])
  mat <- apply(mat, 2, function(x) suppressWarnings(as.numeric(x)))
  if (is.null(dim(mat))) mat <- matrix(mat, ncol = 1L)
  keep <- colMeans(!is.na(mat)) >= 0.7 & apply(mat, 2, stats::sd, na.rm = TRUE) > 0
  mat <- mat[, keep, drop = FALSE]
  if (ncol(mat) == 0) return(dat %>% select(StudyID))
  for (j in seq_len(ncol(mat))) {
    mat[is.na(mat[, j]), j] <- stats::median(mat[, j], na.rm = TRUE)
  }
  pcs <- stats::prcomp(mat, center = TRUE, scale. = TRUE)
  scores <- as.data.frame(pcs$x[, seq_len(min(5L, ncol(pcs$x))), drop = FALSE])
  names(scores) <- paste0(prefix, "_PC", seq_along(scores))
  bind_cols(dat %>% select(StudyID), scores)
}

beta_change_ref <- function(df, exposure, covars, candidate, outcome_pc) {
  need <- unique(c(outcome_pc, exposure, covars))
  need <- need[need %in% names(df)]
  if (!all(c(outcome_pc, exposure, candidate) %in% need)) return(c(NA_real_, NA_integer_, NA_integer_))
  d <- df[, need, drop = FALSE]
  d <- d[stats::complete.cases(d), , drop = FALSE]
  if (nrow(d) < 30) return(c(NA_real_, nrow(d), NA_integer_))

  cov_full <- covars[covars %in% names(d)]
  if (!candidate %in% cov_full) return(c(NA_real_, nrow(d), NA_integer_))
  cov_red <- setdiff(cov_full, candidate)
  f_full <- as.formula(paste(outcome_pc, "~", paste(c(exposure, cov_full), collapse = " + ")))
  f_red <- as.formula(paste(outcome_pc, "~", paste(c(exposure, cov_red), collapse = " + ")))
  m_full <- tryCatch(lm(f_full, data = d), error = function(e) NULL)
  m_red <- tryCatch(lm(f_red, data = d), error = function(e) NULL)
  if (is.null(m_full) || is.null(m_red)) return(c(NA_real_, nrow(d), NA_integer_))
  b_full <- coef(m_full)[exposure]
  b_red <- coef(m_red)[exposure]
  if (is.na(b_full) || is.na(b_red) || !is.finite(b_full) || abs(b_full) < 1e-8) return(c(NA_real_, nrow(model.frame(m_full)), nrow(model.frame(m_red))))
  c(abs((b_red - b_full) / b_full) * 100, nrow(model.frame(m_full)), nrow(model.frame(m_red)))
}

  vars <- vars[vars %in% names(df)]

    Available_N = sapply(vars, function(v) sum(!is.na(df[[v]][idx]))),
    Missing_pct = sapply(vars, function(v) round(mean(is.na(df[[v]][idx])) * 100, 1))
  ) %>% arrange(desc(Missing_pct))
}

4 Model specification

4.1 Proposed models

  1. Visit-specific linear models:
    • Y_v6 ~ E_v6 + core_covars_v6
    • Y_v7 ~ E_v7 + core_covars_v7_lean
  2. Longitudinal conditional model:
    • Y_v7 ~ E_v6 + Y_v6 + baseline_covariates
    • this model is fit separately for each metabolite pair so that visit-7 metabolite values are adjusted for the same metabolite measured at visit 6
  3. Repeated-measures mixed model:
    • Y_it ~ E_it + visit_t + E_it:visit_t + repeated_core_covariates + (1|child)
    • E_it is the visit-harmonized child diet exposure, allowing the same dietary construct to be evaluated across visits 6 and 7
    • sensitivity models additionally harmonize the healthy-diet indicator

4.2 Mathematical model forms

  1. Visit-specific models:

\[ Y_{i,6} = \beta_0 + \beta_1 E_{i,6} + \sum_{k=1}^{p} \gamma_k C^{(6)}_{i,k} + \varepsilon_{i,6} \]

\[ Y_{i,7} = \beta_0 + \beta_1 E_{i,7} + \sum_{k=1}^{q} \delta_k C^{(7)}_{i,k} + \varepsilon_{i,7} \]

  1. Longitudinal conditional model:

\[ Y_{i,7} = \beta_0 + \beta_1 E_{i,6} + \beta_2 Y_{i,6} + \sum_{k=1}^{r} \eta_k B_{i,k} + \varepsilon_i \]

  1. Repeated-measures mixed model (child i, time t):

\[ Y_{i,t} = \beta_0 + \beta_1 E_{i,t} + \beta_2 t + \beta_3(E_{i,t} \times t) + \sum_{k=1}^{s} \theta_k R_{i,t,k} + u_i + \varepsilon_{i,t} \]

with random intercept:

\[ u_i \sim \mathcal{N}(0, \sigma_u^2), \quad \varepsilon_{i,t} \sim \mathcal{N}(0, \sigma^2) \]

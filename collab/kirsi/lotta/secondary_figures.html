fit_cross_models <- function(df, exposure, features, model_sets, analysis_name) {
  bind_rows(lapply(features, function(ft) {
    bind_rows(lapply(names(model_sets), function(model_nm) {
      fit <- fit_lm_exposure(
        df = df,
        outcome = ft,
        exposure = exposure,
        covars = model_sets[[model_nm]]
      )
      if (is.null(fit)) return(NULL)
      fit %>%
        mutate(
          analysis = analysis_name,
          model = model_nm,
          feature = ft
        )
    }))
  }))
}

comp_q1 <- fit_cross_models(
  df = df6,
  exposure = rep_q1,
  features = top_features_q1,
  model_sets = list(
    "Minimally adjusted" = minimal_v6,
    "Primary adjusted" = core_v6,
    "Technical/acute-state adjusted" = technical_v6
  ),
  analysis_name = "Concurrent: 2-year diet and 2-year metabolites"
)

comp_q2 <- fit_cross_models(

fit_prospective_models <- function(df6, df7, exposure, features, core_covars_v6) {
  bind_rows(lapply(features, function(ft) {
    if (!(ft %in% names(df6)) || !(ft %in% names(df7))) return(NULL)

    dat <- inner_join(
      df7 %>% select(StudyID, outcome = all_of(ft)),
      df6 %>% select(StudyID, exposure_value = all_of(exposure), all_of(core_covars_v6), baseline = all_of(ft)),
      by = "StudyID"
    )

    base_covars <- intersect(core_covars_v6, names(dat))
    bind_rows(
      fit_lm_exposure(dat, outcome = "outcome", exposure = "exposure_value", covars = character(0)) %>%
        mutate(model = "Unadjusted"),
      fit_lm_exposure(dat, outcome = "outcome", exposure = "exposure_value", covars = base_covars) %>%
        mutate(model = "Primary adjusted"),
      fit_lm_exposure(dat, outcome = "outcome", exposure = "exposure_value", covars = c(base_covars, "baseline")) %>%
        mutate(model = "Baseline-metabolite adjusted")
    ) %>%
      mutate(feature = ft)
  }))
}

comp_q3 <- fit_prospective_models(

  df7 = df7,
  exposure = rep_q3,
  features = top_features_q3,
  core_covars_v6 = core_v6

}

4 Figure A4. Comparative pattern across Q1-Q3

Show code 
bases_q1 <- unique(strip_visit_suffix(q1$exposure))
bases_q2 <- unique(strip_visit_suffix(q2$exposure))
bases_q3 <- unique(strip_visit_suffix(q3$exposure))
common_bases <- intersect(intersect(bases_q1, bases_q2), bases_q3)

base_score <- bind_rows(
  q1 %>% mutate(base = strip_visit_suffix(exposure)),
  q2 %>% mutate(base = strip_visit_suffix(exposure)),
  q3 %>% mutate(base = strip_visit_suffix(exposure))
) %>%
  filter(base %in% common_bases) %>%
  group_by(base) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  arrange(best_q)

selected_base <- if (nrow(base_score) > 0) base_score$base[[1]] else NA_character_

exp_q1 <- q1 %>% filter(strip_visit_suffix(exposure) == selected_base) %>% distinct(exposure) %>% slice_head(n = 1) %>% pull(exposure)
exp_q2 <- q2 %>% filter(strip_visit_suffix(exposure) == selected_base) %>% distinct(exposure) %>% slice_head(n = 1) %>% pull(exposure)
exp_q3 <- q3 %>% filter(strip_visit_suffix(exposure) == selected_base) %>% distinct(exposure) %>% slice_head(n = 1) %>% pull(exposure)

comp_age <- bind_rows(
  q1 %>% filter(exposure == exp_q1) %>% mutate(model = "Concurrent at 2 years"),
  q2 %>% filter(exposure == exp_q2) %>% mutate(model = "Concurrent at 5-6 years"),
  q3 %>% filter(exposure == exp_q3) %>% mutate(model = "Prospective 2-year to 5-6-year")
) %>%
  add_feature_meta()

top_comp_features <- comp_age %>%
  group_by(feature) %>%
  summarise(best_q = min(q.value, na.rm = TRUE), .groups = "drop") %>%
  slice_min(best_q, n = 20, with_ties = FALSE)

comp_age_plot <- comp_age %>%
  semi_join(top_comp_features, by = "feature") %>%
  mutate(
    feature_lab = paste0(feature_group, " | ", feature),
    model = factor(
      model,
      levels = c("Concurrent at 2 years", "Concurrent at 5-6 years", "Prospective 2-year to 5-6-year")
    )
  )

comp_age_plot %>%
  ggplot(aes(x = feature_lab, y = estimate, color = model)) +
  geom_hline(yintercept = 0, linetype = "dashed", color = "grey45") +
  geom_pointrange(
    aes(ymin = conf.low, ymax = conf.high),

    fatten = 1.4,

class_comp <- bind_rows(
  q1 %>% mutate(analysis = "Q1 concurrent"),
  q2 %>% mutate(analysis = "Q2 concurrent"),
